// 배열 두배 만들기_try1
class Solution {
    public int[] solution(int[] numbers) { 
        int[] numbers = {};
        int numLength = numbers.length;
        int[] copyArr = new int[numLength];

        for(int i = 0; i < numLength + 1; i++) {
            copyArr[i] = numbers[i] * 2;
        }
        
        return copyArr;
}
